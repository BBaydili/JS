function makeRows(m, n) {
    var row = "<th></th>";
    for (var j = 1; j <= n; j++) 
        row += "<th>"+j+"</th>";
    text = "<tr>"+row+"</tr>";
    for (var i = 1; i <= m; i++) {
        row = "<th>"+i+"</th>";
        for (var j = 1; j <= n; j++) {
            var p = n*(i-1)+j;
            var s = "<span class=tip>C"+p+"</span>"
            row += "<td>"+i+","+j+s+"</td>";
        }
        text += "<tr>"+row+"</tr>";
    }
    out.innerHTML = text;
}
